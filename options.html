// options.js document.addEventListener("DOMContentLoaded", () => { chrome.storage.sync.get(["postCount", "scheduleTime"], (data) => { document.getElementById("postCount").value = data.postCount || 3; document.getElementById("scheduleTime").value = data.scheduleTime || "12:00"; }); document.getElementById("save").addEventListener("click", () => { const postCount = parseInt(document.getElementById("postCount").value, 10); const scheduleTime = document.getElementById("scheduleTime").value; chrome.storage.sync.set({ postCount, scheduleTime }); alert("Settings saved successfully!"); }); });
